# Routing

### Overview

The routing module provides mechanisms for handling client-side 
navigation within a single-page application (SPA). It enables 
you to define routes that map URLs to specific components, 
allowing for seamless transitions between different views without 
full page reloads.

### Example

```jsx
import { Router, Route, Link, navigate, useRouter } from "kaioken"

function App() {
  return (
    <Router>
      <Route path="/" element={<HomePage />} />
      <Route path="/login" element={<LoginPage />} />
      <Route path="/users/:userId" element={<UserPage />} />
    </Router>
  )
}

function LoginPage() {
  const user = useUser() // some custom hook that gets auth state
  if (user) {
    return navigate("/")
  }

  return <LoginForm />
}

function UserPage() {
  const { params } = useRouter()
  
  /**
   * the ":userId" part of the Route path is parsed from the URL 
   * and made accessible via useRouter()
   */
  return <h1>User: {params.userId}</h1> 
}
```
